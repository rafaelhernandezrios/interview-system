# Base de Datos MongoDB
# Conecta al cluster "MiraiInnovation" en MongoDB Atlas
# Base de datos: "mirai-interviews", Colección: "users"
# Formato para MongoDB Atlas:
MONGO_URI=mongodb+srv://username:password@miraiinnovation.mongodb.net/mirai-interviews?retryWrites=true&w=majority
# 
# Reemplaza:
# - username: tu usuario de MongoDB Atlas
# - password: tu contraseña de MongoDB Atlas
# - miraiinnovation: nombre de tu cluster (ajusta si es diferente)
# 
# Si usas conexión local en lugar de Atlas:
# MONGO_URI=mongodb://localhost:27017/mirai-interviews

# JWT
JWT_SECRET=your_secret_key_here_change_in_production

# Servidor
PORT=20352
NODE_ENV=development

# CORS
CORS_ORIGINS=http://localhost:3000,http://localhost:3001

# Almacenamiento de CVs y Videos
# Opciones: 'local' (carpeta uploads/) o 's3' (AWS S3)
# Por defecto usa 'local' si no está configurado
# Cuando uses 's3', tanto CVs como Videos se guardarán en S3
STORAGE_TYPE=s3

# AWS S3 (necesario si STORAGE_TYPE=s3)
# Bucket: mirai-interviews
# Carpetas: cvs/ y videos/
AWS_REGION=us-east-1
AWS_ACCESS_KEY_ID=your_key
AWS_SECRET_ACCESS_KEY=your_secret
AWS_BUCKET_NAME=mirai-interviews

# OpenAI
OPENAI_API_KEY=your_openai_key

# Email
EMAIL_USER=your_email@gmail.com
EMAIL_PASS=your_app_password
FRONTEND_URL=http://localhost:3000
